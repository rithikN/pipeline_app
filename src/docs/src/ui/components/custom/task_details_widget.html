 54    def set_image(self, image_path=None):
 55        """
 56        Load an external image into a QPixmap, fit it into a 16:9 placeholder,
 57        and set it to a QLabel.
 58
 59        param image_path: Path to the external image.
 60        """
 61        placeholder_width = 238
 62        placeholder_height = 132
 63
 64        placeholder_pixmap = QPixmap(placeholder_width, placeholder_height)
 65        placeholder_pixmap.fill(Qt.lightGray)
 66
 67        if image_path:
 68            external_pixmap = QPixmap(image_path)
 69
 70            if not external_pixmap.isNull():
 71                scaled_pixmap = external_pixmap.scaled(
 72                    placeholder_width, placeholder_height,
 73                    Qt.KeepAspectRatio, Qt.SmoothTransformation
 74                )
 75
 76                painter = QPainter(placeholder_pixmap)
 77                x_offset = (placeholder_width - scaled_pixmap.width()) // 2
 78                y_offset = (placeholder_height - scaled_pixmap.height()) // 2
 79                painter.drawPixmap(x_offset, y_offset, scaled_pixmap)
 80                painter.end()
 81
 82        self.image_label.setPixmap(placeholder_pixmap)
 83        self._resize_image()
 84
 85    def _update_text_edit(self):
 86        """Update task details text."""
 87        if self._task_details_data:
 88            new_text = "\n".join(
 89                f"{key}: {value}" for key, value in self._task_details_data.items()
 90            )
 91            self.task_details_textEdit.setText(new_text)

 97    def _resize_image(self):
 98        """Maintain 16:9 aspect ratio for the image_label based on preview_frame's width."""
 99        frame_width = self.preview_frame.width()

102            self.image_label.setGeometry(0, 0, frame_width, frame_height)
