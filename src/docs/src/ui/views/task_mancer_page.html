308    def sync_task_selection(self, task_name):
309        """
310        Synchronize task selection between work and review task lists.
311        """
312        if self.sync_in_progress:
313            return

315        self.sync_in_progress = True
316        try:
317            sender = self.sender()

319                self.review_tasklist_widget.set_selected_task(task_name, emit_signal=True)
320            elif sender == self.review_tasklist_widget:

322        finally:
323            self.sync_in_progress = False

576class DataFetchThread(QThread):
577    """
578    Thread responsible for fetching data related to tasks and their statuses.
579    Emit signals on success or error.
580    """
581    data_fetched = Signal(list, dict)
582    error_occurred = Signal(str)
583
584    def __init__(self, project_data):
585        """
586        Initialize the DataFetchThread.
587
588        Args:
589            project_data (dict): Dictionary containing project-related information (e.g., project name).
590        """
591        super().__init__()
592        self.project_data = project_data
593
594    def run(self):
595        """
596        Fetch task data and task status in a background thread. Emit signals upon completion or error.
597        """
598        logger.debug("DataFetchThread started. Fetching task data and task status.")
599        try:
600            task_data = get_taskData(self.project_data)
601            if not task_data:
602                raise Exception("Failed to fetch task data.")
603
604            task_status = get_taskStatus(self.project_data)
605            if not task_status:
606                raise Exception("Failed to fetch task status.")
607
608            logger.debug("Data fetched successfully, emitting data_fetched signal.")
609            self.data_fetched.emit(task_data, task_status)
610        except Exception as e:
611            logger.error(f"Error in DataFetchThread: {e}", exc_info=True)
612            self.error_occurred.emit(str(e))
